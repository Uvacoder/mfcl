## Interactive Example

<Playground>
  <Component initialState={{ value: 1 }}>
    {({ setState, state }) => (
      <>
        <ProgressBar value={state.value} max={15} />
        <Button
          onClick={() => {
            let newValue = state.value + 1
            if (newValue >= 16) {
              newValue = 0
            }
            setState({ value: newValue })
          }}
        >
          Increment
        </Button>
      </>
    )}
  </Component>
</Playground>
