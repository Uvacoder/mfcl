## Basic Usage

<Playground>
  <Component initialState={{ value: 2 }}>
    {({ setState, state }) => (
      <>
        <div>
          Controlled
          <Rating
            name="simple-controlled"
            value={state.value}
            onChange={(event, newValue) => {
              setState({ value: newValue })
            }}
          />
        </div>
        <div>
          Read Only
          <Rating name="read-only" value={3} readOnly />
        </div>
        <div>
          Disabled
          <Rating name="disabled" value={2} disabled />
        </div>
      </>
    )}

  </Component>
</Playground>
