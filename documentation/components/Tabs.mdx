import { Playground, Props } from 'docz'
import Component from 'react-component-component'
import mattressIpsum from 'mattress-ipsum'
import { Tabs, PanelItem, Panel } from '../../src'
import tabData, { customTabHeaders, ipsumArray } from './helpers/TabData.tsx'

# Tabs

```javascript
import { Tabs } from 'mfcl'
```

## Description

Tabs allow you to switch between different views of related information.

## Properties

<Props of={Tabs} />

## Basic Usage

<Playground>
  <Tabs name="default" items={tabData} defaultValue={2} />
</Playground>

## Controlled by state

<Playground>
  <Component initialState={{ activeTab: 0 }}>
    {({ setState, state }) => (
      <>
        <Tabs
          name="controlled"
          onChange={(activeTab) => setState({ activeTab })}
          value={state.activeTab}
          items={tabData}
        />
        <Panel>
          {ipsumArray.map((item, index) => (
            <PanelItem name="controlled" isSelected={index === state.activeTab} index={index} key={index}>
              {item}
            </PanelItem>
          ))}
        </Panel>
      </>
    )}
  </Component>
</Playground>
