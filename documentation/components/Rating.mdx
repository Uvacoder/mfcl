## Controlled Rating

<Playground>
  <Component initialState={{ value: 2 }}>
    {({ setState, state }) => (
      <div>
        Controlled
        <Rating
          name="simple-controlled"
          value={state.value}
          onChange={(event, newValue) => {
            setState({ value: newValue })
          }}
        />
      </div>
    )}
  </Component>
</Playground>
