## Basic Usage

<Playground>
  <Component initialState={{ isOpen: false }}>
    {({ setState, state }) => {
      return (
        <div>
          <Modal
            header="Basic Modal"
            subheader="Subheader Details"
            isOpen={state.isOpen}
            onClose={(isOpen) => setState({ isOpen: false })}
          >
            {mattressIpsum(23)}
            <br />
            <br />
            <Button onClick={() => setState({ isOpen: false })}>Close Modal</Button>
          </Modal>
          <Button btnType="primary" onClick={() => setState({ isOpen: !state.isOpen })} size="lg">
            Open Modal
          </Button>
        </div>
      )
    }}
  </Component>
</Playground>
